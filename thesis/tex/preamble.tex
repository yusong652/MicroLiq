% Safe includegraphics (shows placeholder for missing figures)
\newcommand{\safeincludegraphics}[2][]{%
  \IfFileExists{#2}{%
    \includegraphics[#1]{#2}%
  }{%
    \fcolorbox{gray}{white}{%
      \begin{minipage}[c][0.20\textheight][c]{0.90\linewidth}
        \centering
        \textbf{[Figure placeholder]}\\[4pt]
        \texttt{\detokenize{#2}}
      \end{minipage}
    }%
  }%
}
